position = 0

for i in range from 1 to length(a_list) - 1
    current = i
    key = a_list[current]
    
    if key is greater than or equal to a_list[position] then
        place = binary_loc_finder(a_list, position + 1, current - 1, key)
    else
        place = binary_loc_finder(a_list, 0, position - 1, key)
    end if
    
    position = place
    a_list = place_inserter(a_list, place, current)
end for

function binary_loc_finder(a_list, start, end, key)
    if start is equal to end then
        if a_list[start] is greater than key then
            return start
        else
            return start + 1
    end if
    
    if start is greater than end then
        return start
    end if
    
    middle = (start + end) // 2
    
    if a_list[middle] is less than key then
        return binary_loc_finder(a_list, middle + 1, end, key)
    end if
    
    if a_list[middle] is greater than key then
        return binary_loc_finder(a_list, start, middle - 1, key)
    end if
    
    return middle
end function

function place_inserter(a_list, start, end)

        a_list[k] = a_list[k - 1]
    end for
    
    a_list[start] = temp
    return a_list
end function
